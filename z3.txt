function whatCentury(year) {
    let century = Math.ceil(Number(year) / 100);
    let ending = "th"; 
    if (century % 10 === 1 && century % 100 !== 11) {
        ending = "st";
    } else if (century % 10 === 2 && century % 100 !== 12) {
        ending = "nd";
    } else if (century % 10 === 3 && century % 100 !== 13) {
        ending = "rd";
    }
    return century + ending;
}
