function longest(arr, n) {
    const WithIndex = arr.map((str, index) => ({
        str: str,
        index: index,
        length: str.length
    }));
    WithIndex.sort((a, b) => {
        if (b.length !== a.length) {
            return b.length - a.length; 
        }
        return a.index - b.index; 
    });
    return WithIndex[n - 1].str;
}
