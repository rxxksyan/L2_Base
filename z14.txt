function isCircleSorted( arr ) {
    if (arr.length <= 1) return true;
    let breakPoint = 0;
    for (let i = 0; i < arr.length - 1; i++) {
        if (arr[i] > arr[i + 1]) {
            breakPoint++;
        }
        if (breakPoint > 1) return false;
    }
    if (breakPoint === 1) {
        return arr[arr.length - 1] <= arr[0];
    }
    return true;
}
